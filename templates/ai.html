import requests, time

def tracked_llm_call(prompt, model="gpt-4o", provider="openai"):
    start = time.monotonic()
    # ... your LLM API call ...
    response_text = "..."
    tokens_in, tokens_out = 50, 120
    requests.post("http://YOUR_SOBS_HOST/v1/ai", json={
        "service": "my-app",
        "provider": provider,
        "model": model,
        # Standard OTel GenAI content attributes (recommended)
        "input_messages": [{"role": "user", "content": prompt}],
        "output_messages": [{"role": "assistant", "content": response_text}],
        # Legacy fields also accepted for backward compatibility
        "prompt": prompt,
        "response": response_text,
        "tokens_in": tokens_in,
        "tokens_out": tokens_out,
        "duration_ms": (time.monotonic() - start) * 1000,
    })
